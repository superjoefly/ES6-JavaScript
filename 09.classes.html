Classes

Constructing Objects

ES6 has a new syntax for constructing objects using classes:


      class Car {
        constructor({model}) {
          this.model = model;
        }

        drive() {
          return 'Vroom!';
        }
      }

      let myCar = new Car({model: 'Audi'});

      console.log(myCar); // => Car {model: "Audi"}
      console.log(myCar.model); // => Audi
      console.log(myCar.drive()); // => Vroom!

Extending the Car class:


      // Extend the Car object
      // Has access to properties and methods of the Car object
      class Sedan extends Car {
        constructor(options) {
          super(options); // refers to constructor of parent class
          this.color = options.color;
        }
        honk() {
          return 'Beep!';
        }
      }

      let friendsCar = new Sedan({color: 'red', model: 'Toyota'});

      console.log(friendsCar.model); // => Toyota
      console.log(friendsCar.color); // => red
      console.log(friendsCar.honk()); // => Beep!
      console.log(friendsCar.drive()); // => Vroom!

Monster Game

Another example of using classes:


      class Monster {
        constructor({name}) {
          this.name = name;
          this.health = 100;
        }
      }

      let beast = new Monster({name: 'Beast'});
      console.log(beast.name); // => Beast
      console.log(beast.health); // => 100

      // Extend the Monster object
      class Snake extends Monster {
        constructor(options) {
          super(options);
        }
        bite(monster) {
          return monster.health -= 10;
        }
      }

      let python = new Snake({name: 'Python'});
      console.log(python.name); // => Python
      console.log(python.health); // => 100

      python.bite(beast);
      console.log(beast.health); // => 90
